## 데이터베이스 연동 예시

### TypeScript 타입 정의
```typescript
// src/types/question.types.ts
export interface QuestionSeedData {
  templateCode: string;
  questionNumber: number;
  category: 'vocabulary' | 'reading' | 'grammar' | 'reasoning';
  questionType: 'choice' | 'short_answer' | 'essay';
  questionText: string;
  passage?: string;
  options?: Array<{
    id: number;
    text: string;
  }>;
  correctAnswer: string;
  points: number;
  difficulty: 'easy' | 'medium' | 'hard';
  explanation?: string;
}
```
